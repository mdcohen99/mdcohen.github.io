b1g2024plays <- bigten2024 |> 
  group_by(TeamFull) |> 
  summarise(totalplays = sum(TotalPlays), avgplays = mean(TotalPlays)) |> 
  arrange(desc(totalplays))
            

b1g2025plays <- bigten2025 |> 
  group_by(TeamFull) |> 
  summarise(totalplays = sum(TotalPlays), avgplays = mean(TotalPlays)) |> 
  arrange(desc(totalplays))

combinedplays <- b1g2024plays |> 
  inner_join(b1g2025plays, by = "TeamFull", suffix = c("2024", "2025")) |>
  mutate(combined_total = totalplays2024 + totalplays2025,
         combined_avg = (avgplays2024 + avgplays2025) / 2)
